Flutter doesn't run JavaScript — there's no npm, no CSS variables, no `var()`. But `salt-theme-gen` tokens can reach Flutter through a build-time bridge: a Node.js script that converts OKLCH colors to hex, then writes a Dart constants file that Flutter imports.

The result: your web app and Flutter app use token values derived from the same `generateTheme()` call.

## The concept

```
Node.js build script
  generateTheme({ preset: 'ocean' })
    ↓ OKLCH → hex (culori)
    ↓ write Dart file
  lib/theme/app_tokens.dart
    ↓ import
  AppTheme.buildLight()
    ↓
  MaterialApp(theme: ...)
```

One script, one source of truth.

## Install the bridge dependencies

```bash
npm install salt-theme-gen culori
npm install --save-dev tsx
```

`culori` converts OKLCH to hex — salt-theme-gen outputs OKLCH strings, Flutter's `Color()` needs hex.

## Generation script

```ts
// scripts/generate-flutter-tokens.mts
import { generateTheme } from 'salt-theme-gen';
import { parse, formatHex } from 'culori';
import { writeFileSync, mkdirSync } from 'node:fs';

const theme = generateTheme({ preset: 'ocean' });

function toHex(color: string): string {
  try {
    const parsed = parse(color);
    if (!parsed) return '#000000';
    return formatHex(parsed) ?? '#000000';
  } catch {
    return color.startsWith('#') ? color : '#000000';
  }
}

// Convert hex '#rrggbb' to Flutter '0xFFrrggbb'
function dartColor(hex: string): string {
  return `0xFF${hex.replace('#', '').toUpperCase()}`;
}

function dartColorConst(name: string, hex: string): string {
  return `  static const Color ${name} = Color(${dartColor(hex)});`;
}

function colorsClass(mode: typeof theme.light, className: string): string {
  const entries = Object.entries(mode.colors)
    .map(([k, v]) => {
      const name = k.replace(/([A-Z])/g, m => m.toLowerCase())
                    .replace(/-([a-z])/g, (_, c) => c.toUpperCase());
      return dartColorConst(name, toHex(v));
    });

  return `class ${className} {\n${entries.join('\n')}\n}`;
}

function spacingClass(mode: typeof theme.light): string {
  const entries = Object.entries(mode.spacing)
    .map(([k, v]) => `  static const double ${k} = ${v}.0;`);
  return `class AppSpacing {\n${entries.join('\n')}\n}`;
}

function radiusClass(mode: typeof theme.light): string {
  const entries = Object.entries(mode.radius)
    .map(([k, v]) => `  static const double ${k} = ${v}.0;`);
  return `class AppRadius {\n${entries.join('\n')}\n}`;
}

function fontSizeClass(mode: typeof theme.light): string {
  const entries = Object.entries(mode.fontSizes)
    .map(([k, v]) => {
      const key = k === '3xl' ? 'xxxl' : k;
      return `  static const double ${key} = ${v}.0;`;
    });
  return `class AppFontSize {\n${entries.join('\n')}\n}`;
}

const dart = [
  '// Generated by salt-theme-gen — do not edit manually',
  '// Run: npm run generate:tokens',
  "import 'package:flutter/material.dart';",
  '',
  colorsClass(theme.light, 'AppColorsLight'),
  '',
  colorsClass(theme.dark,  'AppColorsDark'),
  '',
  spacingClass(theme.light),
  '',
  radiusClass(theme.light),
  '',
  fontSizeClass(theme.light),
].join('\n');

mkdirSync('lib/theme', { recursive: true });
writeFileSync('lib/theme/app_tokens.dart', dart, 'utf8');
console.log('✓ lib/theme/app_tokens.dart written');
```

```json
// package.json

  "scripts": {
    "generate:tokens": "tsx scripts/generate-flutter-tokens.mts"
  }
}
```

## Flutter ThemeData

```dart
// lib/theme/app_theme.dart
import 'package:flutter/material.dart';
import 'app_tokens.dart';

class AppTheme {
  static ThemeData buildLight() => ThemeData(
    brightness:   Brightness.light,
    primaryColor: AppColorsLight.primary,
    scaffoldBackgroundColor: AppColorsLight.background,
    colorScheme: ColorScheme.light(
      primary:    AppColorsLight.primary,
      secondary:  AppColorsLight.secondary,
      background: AppColorsLight.background,
      surface:    AppColorsLight.surface,
      error:      AppColorsLight.danger,
      onPrimary:  AppColorsLight.onPrimary,
      onError:    AppColorsLight.onDanger,
    ),
    cardTheme: CardTheme(
      color:        AppColorsLight.surface,
      shape:        RoundedRectangleBorder(
        borderRadius: BorderRadius.circular(AppRadius.lg),
        side:         BorderSide(color: AppColorsLight.border),
      ),
    ),
    textTheme: TextTheme(
      bodyMedium: TextStyle(
        color:    AppColorsLight.text,
        fontSize: AppFontSize.md,
      ),
    ),
  );

  static ThemeData buildDark() => ThemeData(
    brightness:   Brightness.dark,
    primaryColor: AppColorsDark.primary,
    scaffoldBackgroundColor: AppColorsDark.background,
    // ...
  );
}
```

```dart
// lib/main.dart
import 'package:flutter/material.dart';

void main() => runApp(const MyApp());

  const MyApp({super.key});

  @override

    return MaterialApp(
      theme:      AppTheme.buildLight(),

      home: const HomeScreen(),
    );
  }
}
```

## CI pipeline

```yaml
# .github/workflows/build.yml
- name: Generate tokens
  run: npm run generate:tokens

- name: Build Flutter
  run: flutter build apk
```

Token generation runs before the Flutter build. If the Node.js script fails (invalid theme, missing package), the Flutter build never starts.

## Keeping web and Flutter in sync

Both the web app and the Flutter app call the same `generateTheme()` with the same options. Change the preset in `scripts/generate-flutter-tokens.mts` and the web `src/theme.ts` simultaneously to keep them synchronized.

For a shared config:

```ts
// theme-config.json
{ "preset": "ocean", "spacing": "default", "radius": "default", "fontSize": "default" }
```

Both scripts read from this file:

```ts
import config from '../theme-config.json';
const theme = generateTheme(config);
```
